# 交叉编译arm
```bash
sudo dpkg --add-architecture arm64
sudo apt update

apt install -y libdrm-dev:arm64

# 安装字体支持库
sudo apt-get install libfreetype-dev:arm64
# curl
apt install libcurl4-openssl-dev:arm64

apt install libqrencode4:arm64 libqrencode-dev:arm64

apt install libjsoncpp-dev:arm64

apt install libwebp-dev:arm64
apt install libssl-dev:arm64

apt install libpaho-mqttpp-dev:arm64
apt install libpaho-mqtt-dev:arm64

apt install -y libjpeg-dev:arm64 libpng-dev:arm64 curl:arm64



mkdir build-aarch64 && cd build-aarch64
cmake -DCMAKE_TOOLCHAIN_FILE=../aarch64-toolchain.cmake ..
make -j$(nproc)


```
